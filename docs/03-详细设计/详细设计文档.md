class LLMClient:
    """LLM统一接入客户端"""
    
    def __init__(self, config: dict):
        self.api_key = config['api_key']
        self.base_url = config['base_url']
        self.model = config['model']
        self.client = OpenAI(
            api_key=self.api_key,
            base_url=self.base_url
        )
    
    def chat(self, messages: list, temperature: float = 0.7) -> str:
        """
        发送对话请求
        
        Args:
            messages: [{"role": "user/assistant", "content": "..."}]
            temperature: 0-2, 越低越确定性
        
        Returns:
            str: LLM回复内容
        """
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=temperature
        )
        return response.choices[0].message.content
```

# src/image_analyzer.py
from dataclasses import dataclass
from typing import Optional
import cv2
import numpy as np

@dataclass
class AnalysisResult:
    """分析结果数据类"""
    file_path: str
    width: int
    height: int
    format: str
    size_bytes: int
    histogram: Optional[dict] = None
    dynamic_range: Optional[dict] = None
    noise_level: Optional[float] = None
    color_analysis: Optional[dict] = None
    exif: Optional[dict] = None

class ImageAnalyzer:
    """图像分析器"""
    
    SUPPORTED_FORMATS = ['jpg', 'jpeg', 'png', 'tiff', 'tif', 'dng']
    MAX_SIZE = 50 * 1024 * 1024  # 50MB
    
    def analyze(self, image_path: str) -> AnalysisResult:
        """执行完整分析"""
        # 1. 校验文件
        self._validate(image_path)
        
        # 2. 读取图像
        img = cv2.imread(image_path)
        if img is None:
            raise ValueError(f"无法读取图像: {image_path}")
        
        # 3. 提取信息
        result = AnalysisResult(
            file_path=image_path,
            width=img.shape[1],
            height=img.shape[0],
            format=Path(image_path).suffix[1:],
            size_bytes=Path(image_path).stat().st_size
        )
        
        # 4. 直方图分析
        result.histogram = self.analyze_histogram(img)
        
        # 5. 动态范围
        result.dynamic_range = self.analyze_dynamic_range(img)
        
        # 6. 噪声分析
        result.noise_level = self.analyze_noise(img)
        
        # 7. 色彩分析
        result.color_analysis = self.analyze_color(img)
        
        return result
    
    def analyze_histogram(self, img: np.ndarray) -> dict:
        """分析直方图"""
        result = {}
        for i, color in enumerate(['B', 'G', 'R']):
            hist = cv2.calcHist([img], [i], None, [256], [0, 256])
            result[color] = hist.flatten().tolist()
        return result
    
    def analyze_dynamic_range(self, img: np.ndarray) -> dict:
        """分析动态范围"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        return {
            'min': int(gray.min()),
            'max': int(gray.max()),
            'range': int(gray.max() - gray.min())
        }
    
    def analyze_noise(self, img: np.ndarray) -> float:
        """估计噪声水平"""
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        # 使用LAV方法估计噪声
        sigma = np.std(gray)
        return float(sigma)
    
    def analyze_color(self, img: np.ndarray) -> dict:
        """分析色彩"""
        # 计算各通道均值
        b, g, r = cv2.split(img)
        return {
            'B_mean': float(b.mean()),
            'G_mean': float(g.mean()),
            'R_mean': float(r.mean()),
            'white_balance': 'OK' if abs(b.mean() - g.mean()) < 10 else '偏色'
        }
```

# src/qa_engine.py
class QAEngine:
    """智能问答引擎"""
    
    SYSTEM_PROMPT = """你是一位专业的ISP图像调试工程师助手。
你的职责：
1. 解读图像分析结果
2. 提供ISP参数调整建议
3. 解释ISP相关技术概念
4. 帮助用户优化图像质量

请用中文回复，保持专业但易懂。"""
    
    def __init__(self, llm_client: LLMClient):
        self.llm = llm_client
        self.conversation_history = []
    
    def chat(self, user_input: str, context: dict = None) -> str:
        """处理用户问答"""
        messages = [{"role": "system", "content": self.SYSTEM_PROMPT}]
        
        # 添加历史对话
        messages.extend(self.conversation_history[-10:])
        
        # 添加当前输入
        messages.append({"role": "user", "content": user_input})
        
        # 添加上下文（如有）
        if context:
            context_str = f"\n\n当前图像分析结果:\n{json.dumps(context, ensure_ascii=False)}"
            messages[-1]["content"] += context_str
        
        response = self.llm.chat(messages)
        
        # 记录历史
        self.conversation_history.append({"role": "user", "content": user_input})
        self.conversation_history.append({"role": "assistant", "content": response})
        
        return response
    
    def generate_suggestion(self, analysis_result: AnalysisResult) -> str:
        """根据分析结果生成调优建议"""
        prompt = f"""基于以下图像分析结果，请给出ISP参数调整建议：

- 分辨率: {analysis_result.width}x{analysis_result.height}
- 动态范围: {analysis_result.dynamic_range}
- 噪声水平: {analysis_result.noise_level}
- 色彩分析: {analysis_result.color_analysis}

请列出具体的参数调整方向（如：降噪力度、白平衡偏移等）"""

        messages = [
            {"role": "system", "content": self.SYSTEM_PROMPT},
            {"role": "user", "content": prompt}
        ]
        
        return self.llm.chat(messages)

```python
# src/utils/logger.py
import logging
from pathlib import Path

def setup_logger(name: str, log_file: str = None):
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)
    
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    
    # 控制台输出
    console = logging.StreamHandler()
    console.setFormatter(formatter)
    logger.addHandler(console)
    
    # 文件输出
    if log_file:
        Path(log_file).parent.mkdir(parents=True, exist_ok=True)
        file_handler = logging.FileHandler(log_file)
        file_handler.setFormatter(formatter)
        logger.addHandler(file_handler)
    
    return logger
```
