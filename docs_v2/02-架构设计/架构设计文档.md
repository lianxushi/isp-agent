### 4.2 Python Bridge 设计

```typescript
// Python 进程管理
class PythonBridge {
  private pythonPath: string;
  private process: ChildProcess | null = null;
  private socketPath: string;
  
  async execute(script: string, args: object): Promise<any> {
    // 1. 启动 Python 进程 (如未启动)
    await this.ensureProcess();
    
    // 2. 通过 Unix Socket 发送请求
    const response = await this.sendRequest({
      script,
      args,
      requestId: uuid()
    });
    
    return response;
  }
  
  async ensureProcess(): Promise<void> {
    if (!this.process) {
      this.process = spawn(this.pythonPath, [this.scriptPath], {
        stdio: ['pipe', 'pipe', 'pipe']
      });
    }
  }
}
```

```dockerfile
# 安装系统依赖
RUN apt-get update && apt-get install -y \
    libopencv-dev \
    ffmpeg \
    && rm -rf /var/lib/apt/lists/*

# 安装 Python 依赖
COPY requirements.txt /tmp/
RUN pip install -r /tmp/requirements.txt

# 复制 SDK
COPY python/ /opt/isp-sdk/

# 设置工作目录
WORKDIR /app

# 默认命令
CMD ["python", "-m", "isp_sdk.server"]
```

## 七、配置设计

### 7.1 配置文件

```yaml
# config.yaml
isp:
  python_path: /usr/bin/python3
  sdk_path: /opt/isp-sdk
  
  # LLM 配置
  llm:
    provider: minimax  # minimax / openai / anthropic
    model: MiniMax-M2.5
    api_key: ${ISP_LLM_API_KEY}
  
  # 知识库配置
  knowledge_base:
    type: qdrant  # qdrant / lancedb / memory
    path: /data/isp-kb
    collection: isp_cases
  
  # 分析配置
  analysis:
    max_image_size: 50MB
    max_video_duration: 300s
    timeout: 30s

# OpenClaw Agent 配置
agent:
  id: isp-agent
  name: ISP智能助手
  model: MiniMax-M2.5
  skills:
    - isp_tuning
    - isp_qa
    - isp_auto_driving
    - isp_knowledge
```
